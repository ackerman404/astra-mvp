def _on_ingestion_progress(self, info: dict):
    if info["stage"] == "scanning":
        self.startup_screen.set_status(f"Found {info['total_files']} files")
        self.startup_screen.show_progress_bar(True)
        self.startup_screen.set_progress(0, info['total_files'])
    elif info["stage"] == "processing":
        name = info["current_file_name"]
        idx = info["current_file_index"] + 1
        total = info["total_files"]
        self.startup_screen.set_status(f"Processing {name} ({idx} of {total})")
        self.startup_screen.set_progress(idx, total)
